Antes de fazer qualquer modificação foi preciso atulaizar o projeto, criar uma branch

```bash
# Atualizar o projeto
git pull

# Criar uma branch
git checkout -b <name-of-your-patch-branch> origin/testing

# Fazer um backup caso algo dê errado
make nconfig
cp .config .config.mpu6050_working.bak
mkdir -p ~/kernel_backups
cp .config ~/kernel_backups/config_iio_mpu6050_$(date +%F)
```

Agora podemos escolher o que queremos fazer no kernel. Após fazer todas as alterações vamos para o envio do patch.

```bash
make <caminho-do-diretório>
git add <caminho-do-arquivo-ou-diretório>

# Verifica se estão apenas os arquivos modificados
git status
git diff --cache # arquivos add para commit

# Commita assinado
git commit -s

# Procura por problemas nos arquivos commitados
scripts/checkpatch.pl --git HEAD

# Simulando envio do patch
kw send-patch --send -1 --private --simulate --to='<EMAIL-ADDRESS-1>','<EMAIL-ADDRESS-2>' ...
```

## Problemas

### Lista de Commit Com Arquivos Inesperados

Durante o commit descobri que tinha arquivos que não queria commitar na lista. Quando fui checar quais foram as modificações, eram em sua maioria mudança de modo de mod 100644 para mode 100755, então executei um comando que ignora esse tipo de alteração e limpei a lista de commit. Depois adicionai apenas o que queria.

```bash
# Ignora mudança de permissão
git config core.fileMode false 

# Remove todos os adds mas não mexe nos arquivos alterados
git reset
```

### Erro de Estilo

Ao executar o `checkpatch` o sistema retornou vários problemas e alertas, em grande maioria sobre espaçamento utilizando `space` ou invés de `tab`. Portanto utilizei o seguinte comando que corrigi automaticamente os erros de estilo do arquivo.

ATENÇÃO: o comando pode alterar a semântica e a lagibilidade do arquivo. Utilize com **cautela**.

```bash
scripts/checkpatch.pl -f <caminho-do-arquivo> --fix-inplace
```

	switch (mask) {

			*val  = 0;

			mutex_unlock(&st->lock);

			return IIO_VAL_INT_PLUS_NANO;
		case IIO_ACCEL:
			mutex_lock(&st->lock);
			*val = 0;
			*val2 = accel_scale[st->chip_config.accl_fs];
			mutex_unlock(&st->lock);

			return IIO_VAL_INT_PLUS_MICRO;
		case IIO_TEMP:
			*val = st->hw->temp.scale / 1000000;
			*val2 = st->hw->temp.scale % 1000000;
			return IIO_VAL_INT_PLUS_MICRO;
		case IIO_MAGN:
			return inv_mpu_magn_get_scale(st, chan, val, val2);
		default:
			return -EINVAL;
		}
	case IIO_CHAN_INFO_OFFSET:
		switch (chan->type) {
		case IIO_TEMP:
			*val = st->hw->temp.offset;
			return IIO_VAL_INT;
		default:
			return -EINVAL;
		}
	case IIO_CHAN_INFO_CALIBBIAS:
		switch (chan->type) {
		case IIO_ANGL_VEL:
			mutex_lock(&st->lock);
			ret = inv_mpu6050_sensor_show(st, st->reg->gyro_offset,
						chan->channel2, val);
			mutex_unlock(&st->lock);
			return ret;
		case IIO_ACCEL:
			mutex_lock(&st->lock);
			ret = inv_mpu6050_sensor_show(st, st->reg->accl_offset,
						chan->channel2, val);
			mutex_unlock(&st->lock);
			return ret;

		default:
			return -EINVAL;
		}
	default:
		return -EINVAL;
	}
